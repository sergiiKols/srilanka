## ðŸŽ¯ Key Metrics to Track

```sql
-- Daily Active Users
SELECT COUNT(DISTINCT user_id) 
FROM analytics_events 
WHERE created_at > NOW() - INTERVAL '1 day';

-- Conversion Rate (Request â†’ Offer â†’ Contact)
SELECT 
  COUNT(DISTINCT rr.id) as total_requests,
  COUNT(DISTINCT o.id) as total_offers,
  COUNT(DISTINCT m.id) as total_messages,
  ROUND(COUNT(DISTINCT m.id)::numeric / COUNT(DISTINCT rr.id) * 100, 2) as conversion_rate
FROM rental_requests rr
LEFT JOIN offers o ON o.request_id = rr.id
LEFT JOIN messages m ON m.offer_id = o.id
WHERE rr.created_at > NOW() - INTERVAL '30 days';

-- Average Response Time
SELECT AVG(EXTRACT(EPOCH FROM (o.created_at - rr.created_at)) / 3600) as avg_hours
FROM offers o
JOIN rental_requests rr ON rr.id = o.request_id
WHERE o.created_at > NOW() - INTERVAL '30 days';
```
